function render() { const card = document.getElementById('rosterCard'); // Dynamic refresh animation container card.classList.remove('run-animation'); const day = document.getElementById('daySelect').value; const dateVal = document.getElementById('dateInput').value; const data = rosterData[day]; if (!data) return; const morningSet = new Set(data.morning); const eveningSet = new Set(data.evening); let tableRows = ''; let morningTotal = data.morning.length; let eveningTotal = data.evening.length; let index = 0; // সব বাচ্চাদের নামের তালিকা ধরে একবারই লুপ চলবে ALL_CHILDREN.forEach(name => { const isMorning = morningSet.has(name); const isEvening = eveningSet.has(name); let delay = (index * 0.03) + 's'; // কোনো বাচ্চা যদি অন্তত একটি শিফটেও ডিউটি পায়, তবেই তাকে তালিকায় দেখাবে if (isMorning || isEvening) { let rowClass = ""; let mValue = ""; let eValue = ""; if (isMorning && isEvening) { rowClass = "row-both"; mValue = "1"; eValue = "1"; } else if (isMorning) { rowClass = "row-morning"; mValue = "1"; eValue = ""; } else if (isEvening) { rowClass = "row-evening"; mValue = ""; eValue = "1"; } tableRows += ` ${name} ${mValue} ${eValue} `; index++; } }); let leaveRows = ''; const leaveEntries = Object.entries(data.leave); if (leaveEntries.length === 0) { leaveRows = `No leave data`; } else { leaveEntries.forEach(([name, info]) => { leaveRows += `${name}${info}`; }); } const offdayNames = data.offday.length > 0 ? data.offday.join(', ') : '—'; const dateDisplay = dateVal ? formatDisplayDate(dateVal) : day; card.innerHTML = `
CHILDREN
${tableRows}
NameMorningEvening
0
Total ${morningTotal} ${eveningTotal}
TOTAL= ${morningTotal + eveningTotal}
OFFDAY
${leaveRows}
DepName
CHILDREN${offdayNames}
Leave Data
Trail
Morning
Evening
${dateDisplay}
`; // Trigger smooth rendering layout after nodes injection setTimeout(() => { card.classList.add('run-animation'); }, 10); }
